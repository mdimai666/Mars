# VarNode

Declared varibles

# Support types

- ["int"] = typeof(int),
- ["long"] = typeof(long),
- ["float"] = typeof(float),
- ["double"] = typeof(double),
- ["decimal"] = typeof(decimal),
- ["bool"] = typeof(bool),
- ["string"] = typeof(string),
- ["DateTime"] = typeof(DateTime),
- ["Guid"] = typeof(Guid),

- [typeof(int[])] = "int[]",
- [typeof(long[])] = "long[]",
- [typeof(float[])] = "float[]",
- [typeof(double[])] = "double[]",
- //[typeof(decimal[])] = "decimal[]", not work on C#
- [typeof(bool[])] = "bool[]",
- [typeof(string[])] = "string[]",
